\subsection*{Aggregation Pipelines (executed concurrently via Promise.all)}
\begin{enumerate}
  \item \textbf{Total Revenue} --- Sum of \texttt{totalPrice} for non-cancelled orders.
  \item \textbf{This Month Revenue} --- Same, filtered by current month start.
  \item \textbf{Last Month Revenue} --- Same, filtered by last month boundaries.
  \item \textbf{Orders by Status} --- Group by \texttt{status}, count each.
  \item \textbf{Top 5 Products} --- Unwind items, group by product, sort by \texttt{totalSold} descending.
  \item \textbf{Daily Revenue (7 days)} --- Group by date string, sum revenue and count orders.
  \item \textbf{Recent Orders} --- Last 5 orders, populated with user info.
  \item \textbf{Counts} --- \texttt{totalCustomers}, \texttt{totalAdmins}, \texttt{totalProducts}.
  \item \textbf{Growth Percent} --- \texttt{((thisMonth - lastMonth) / lastMonth) * 100}.
\end{enumerate}
