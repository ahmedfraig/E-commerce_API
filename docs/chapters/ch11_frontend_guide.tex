\subsection{Credentials \& Cookies}

The API stores the \texttt{refreshToken} in an \textbf{HttpOnly Cookie}. Your HTTP client must be configured to send cookies:

\begin{lstlisting}[language=JavaScript,title=Axios Setup (Recommended)]
import axios from 'axios';

const api = axios.create({
  baseURL: 'http://localhost:5000',
  withCredentials: true,  // CRITICAL: sends/receives cookies
  headers: {
    'Content-Type': 'application/json'
  }
});

export default api;
\end{lstlisting}

\begin{lstlisting}[language=JavaScript,title=Fetch API Alternative]
fetch('http://localhost:5000/auth/login', {
  method: 'POST',
  credentials: 'include',  // CRITICAL
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify({ email, password })
});
\end{lstlisting}

\section{Authentication Flow}
% ══════════════════════════════════════════════════════════════════════════

\subsection{Registration Flow}

\begin{enumerate}
  \item \textbf{Step 1:} Call \texttt{POST /auth/register/send-otp} with \texttt{\{username, email, password\}}.
  \item \textbf{Step 2:} User receives 6-digit OTP via email.
  \item \textbf{Step 3:} Call \texttt{POST /auth/verify-otp} with \texttt{\{email, otp\}}.
  \item \textbf{Step 4:} API returns \texttt{accessToken} + sets \texttt{refreshToken} cookie. User is now logged in.
\end{enumerate}

\begin{lstlisting}[language=JavaScript,title=Registration Implementation]
// Step 1: Send OTP
const registerUser = async (userData) => {
  const res = await api.post('/auth/register/send-otp', {
    username: userData.username,
    email: userData.email,
    password: userData.password,
    phone: userData.phone // optional
  });
  return res.data; // { success: true, message: "OTP sent to email" }
};

// Step 2: Verify OTP
const verifyOtp = async (email, otp) => {
  const res = await api.post('/auth/verify-otp', { email, otp });
  // Store the access token
  localStorage.setItem('accessToken', res.data.accessToken);
  return res.data; // { success, accessToken, user }
};
\end{lstlisting}

\subsection{Login Flow}

\begin{lstlisting}[language=JavaScript,title=Login Implementation]
const login = async (email, password) => {
  const res = await api.post('/auth/login', { email, password });
  localStorage.setItem('accessToken', res.data.accessToken);
  return res.data.user;

\end{lstlisting}

\subsection{Setting the Authorization Header}

\begin{lstlisting}[language=JavaScript,title=Request Interceptor]
api.interceptors.request.use((config) => {
  const token = localStorage.getItem('accessToken');
  if (token) {
    config.headers.Authorization = `Bearer ${token}`;
  }
  return config;
});
\end{lstlisting}

\subsection{Silent Token Refresh (Interceptor Pattern)}

The \texttt{accessToken} expires after \textbf{15 minutes}. Build an Axios interceptor for automatic, silent renewal:

\begin{lstlisting}[language=JavaScript,title=Response Interceptor for Token Refresh]
let isRefreshing = false;
let failedQueue = [];

const processQueue = (error, token = null) => {
  failedQueue.forEach(prom => {
    if (error) prom.reject(error);
    else prom.resolve(token);
  });
  failedQueue = [];
};

api.interceptors.response.use(
  (response) => response,
  async (error) => {
    const originalRequest = error.config;

    if (error.response?.status === 401
        && !originalRequest._retry) {
      if (isRefreshing) {
        return new Promise((resolve, reject) => {
          failedQueue.push({ resolve, reject });
        }).then(token => {
          originalRequest.headers.Authorization
            = `Bearer ${token}`;
          return api(originalRequest);
        });
      }

      originalRequest._retry = true;
      isRefreshing = true;

      try {
        // Cookie is sent automatically
        const res = await api.get('/auth/refresh-token');
        const newToken = res.data.accessToken;
        localStorage.setItem('accessToken', newToken);
        api.defaults.headers.common.Authorization
          = `Bearer ${newToken}`;
        processQueue(null, newToken);
        originalRequest.headers.Authorization
          = `Bearer ${newToken}`;
        return api(originalRequest);
      } catch (err) {
        processQueue(err, null);
        localStorage.removeItem('accessToken');
        window.location.href = '/login';
        return Promise.reject(err);
      } finally {
        isRefreshing = false;
      }
    }
    return Promise.reject(error);
  }
);
\end{lstlisting}

\subsection{Password Reset Flow}
\begin{enumerate}
  \item Call \texttt{POST /auth/forgot-password/send-otp} with \texttt{\{email\}}.
  \item User clicks the link in their email: \texttt{FRONTEND\_URL/reset-password/<token>}.
  \item Frontend extracts the token from the URL.
  \item Call \texttt{POST /auth/forgot-password/reset} with \texttt{\{token, newPassword\}}.
\end{enumerate}

\begin{lstlisting}[language=JavaScript,title=Reset Password Page]
// URL: /reset-password/:token
const ResetPasswordPage = () => {
  const token = useParams().token;
  const handleSubmit = async (newPassword) => {
    const res = await api.post(
      '/auth/forgot-password/reset',
      { token, newPassword }
    );
    localStorage.setItem(
      'accessToken', res.data.accessToken
    );
    // User is now logged in with new password
  };
};
\end{lstlisting}

// Get all products with filters
const getProducts = async (filters = {}) => {
  const params = new URLSearchParams(filters);
  const res = await api.get(`/products?${params}`);
  return res.data;
};

// Search products
const searchProducts = async (query) => {
  const res = await api.get('/products/search', {
    params: {
      text: query.text,
      category: query.category,
      minPrice: query.minPrice,
      maxPrice: query.maxPrice,
      sort: query.sort,
      page: query.page
    }
  });
  return res.data;
};

// Get single product
const getProduct = async (id) => {
  const res = await api.get(`/products/${id}`);
  return res.data.data;
};

// Admin: Create product (multipart/form-data)
const createProduct = async (productData, imageFiles) => {
  const formData = new FormData();
  Object.keys(productData).forEach(key => {
    formData.append(key, productData[key]);
  });
  imageFiles.forEach(file => {
    formData.append('images', file);
  });
  const res = await api.post('/products', formData, {
    headers: { 'Content-Type': 'multipart/form-data' }
  });
  return res.data;
};

// Add review (must have purchased + received product)
const addReview = async (productId, rating, comment) => {
  const res = await api.post(
    `/products/${productId}/reviews`,
    { rating, comment }
  );
  return res.data;
};
\end{lstlisting}

\section{Error Handling}
% ══════════════════════════════════════════════════════════════════════════

\begin{lstlisting}[language=JavaScript,title=Global Error Handler]
api.interceptors.response.use(
  (response) => response,
  (error) => {
    const message = error.response?.data?.message
      || 'An unexpected error occurred';
    const status = error.response?.status;

    // Handle specific status codes
    switch (status) {
      case 400: // Validation error
        showToast(message, 'error');
        break;
      case 401: // Unauthorized (handled by refresh)
        break;
      case 403: // Forbidden
        showToast('Access denied', 'error');
        break;
      case 404: // Not found
        showToast(message, 'warning');
        break;
      case 429: // Rate limited
        showToast(
          'Too many requests. Please wait.', 'warning'
        );
        break;
      default:
        showToast(message, 'error');
    }

    return Promise.reject(error);
  }
);
\end{lstlisting}
